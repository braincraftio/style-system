Design Tokens Usage

/* Using Aurora Design Tokens */
.custom-component {
  /* Colors */
  background: var(--color-surface-paper);
  color: var(--color-text-primary);
  border: 1px solid var(--color-border-subtle);

  /* Spacing */
  padding: var(--space-4);
  margin-bottom: var(--space-6);
  gap: var(--space-3);

  /* Typography */
  font-size: var(--text-base);
  line-height: var(--leading-relaxed);

  /* Border Radius */
  border-radius: var(--radius-lg);

  /* Shadows */
  box-shadow: var(--shadow-sm);

  /* Transitions */
  transition: all var(--duration-normal) var(--ease-spring);
}

/* Hover State */
.custom-component:hover {
  transform: translateY(var(--translate-hover));
  box-shadow: var(--shadow-md);
}

React Component

import { useState } from 'react';
import { Button, Card, Badge } from '@aurora/react';

function FeatureCard({ title, description, status }) {
  const [isHovered, setIsHovered] = useState(false);

  return (
    <Card
      variant={isHovered ? 'interactive' : 'default'}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
    >
      <Card.Header>
        <h3>{title}</h3>
        <Badge variant={status === 'new' ? 'primary' : 'secondary'}>
          {status}
        </Badge>
      </Card.Header>
      <Card.Body>
        <p>{description}</p>
      </Card.Body>
      <Card.Footer>
        <Button variant="primary" size="sm">
          Learn More
        </Button>
      </Card.Footer>
    </Card>
  );
}

Accessibility Patterns

<!-- Accessible Modal Pattern -->
<div
  class="modal"
  role="dialog"
  aria-modal="true"
  aria-labelledby="modal-title"
  aria-describedby="modal-description"
>
  <div class="modal-backdrop" aria-hidden="true"></div>
  <div class="modal-content">
    <div class="modal-header">
      <h2 id="modal-title">Confirm Action</h2>
      <button
        class="btn btn-icon btn-ghost"
        aria-label="Close modal"
        onclick="closeModal()"
      >
        <svg aria-hidden="true">...</svg>
      </button>
    </div>
    <div class="modal-body">
      <p id="modal-description">
        Are you sure you want to proceed with this action?
      </p>
    </div>
    <div class="modal-footer">
      <button class="btn btn-ghost" onclick="closeModal()">
        Cancel
      </button>
      <button class="btn btn-primary" onclick="confirmAction()">
        Confirm
      </button>
    </div>
  </div>
</div>

Design Tokens

Aurora's design tokens are the foundation of our system. They ensure consistency across all platforms and enable dynamic theming.

Installation

npm install @aurora/design-tokens
# or
yarn add @aurora/design-tokens

Usage

// Import tokens
import { tokens } from '@aurora/design-tokens';

// Use in CSS-in-JS
const styles = {
  background: tokens.colors.surface.paper,
  color: tokens.colors.text.primary,
  padding: tokens.spacing[4],
  borderRadius: tokens.radius.lg
};
